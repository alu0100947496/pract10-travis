def impactoCarbono
	a = @alim.collect{|valor| valor.impacto}
	#b = @cant.collect{|valor| valor}
	b = a.reduce(0){|sum, num| sum + num}
	
	if b < 800
		1
	elsif b < 1200
		2
	else
		3
	end
end

def totalAporte
	a = impactoCarbono
	b = valorCalorico
	c = ((a+b)/2).truncate(0)
	c
end

def valorCalorico
	a = @alim.collect{|valor| valor.valorEnergetico}
	#b = @cant.collect{|valor| valor}
	b = a.reduce(0){|sum, num| sum + num}

	if b < 670
		1
	elsif b < 830
		2
	else
		3
	end
end
