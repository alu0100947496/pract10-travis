def delete(valor)
    current_node = Node.new(nil,nil,valor)
    node = Node.new(nil,nil,valor)
    current_node.next = @head

    if current_node.value == @head.value
        @head = current_node.next
    else
        while (current_node.next != nil) && (current_node.next.value != node.value)
            current_node = current_node.next
        end
        if current_node.next != nil
            current_node.next = current_node.next.next
            @tail = @tail.prev
        end
    end
end

def each
    current_node = @head

    yield current_node.value

    while current_node.next != nil
        current_node = current_node.next

        yield current_node.value
    end
end

def getValor(n)
    if n == 1
        @head.value
    elsif @head.next != nil            
        current_node = @head.next
        i = 2
        while i < n
            current_node = current_node.next
            i += 1
        end

        if current_node != nil
            current_node.value
        else
            nil
        end
    else
        nil
    end
end

def insert(valor)
    nodo = Node.new(nil, nil, valor)

    if @head == nil and @tail == nil
        @tail = nodo
        @head = nodo
    else
        auxNode = @tail
        auxNode.next = nodo
        nodo.prev = auxNode

        @tail = nodo
    end
end

def size
    if @head != nil
        current_node = @head

        size = 1

        while current_node.next != nil
            current_node = current_node.next

            size += 1
        end

        size
    else
        return 0
    end
end

def valor
    current_node = @head

    if current_node.value.class == Integer
        valor = current_node.value

        while current_node.next != nil
            current_node = current_node.next

            valor += current_node.value
        end

        return valor
    else
        return nil
    end
end
